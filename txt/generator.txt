from sdv.metadata import Metadata
from sdv.single_table import CTGANSynthesizer, TVAESynthesizer, CopulaGANSynthesizer
from sklearn.model_selection import train_test_split
from opacus import PrivacyEngine
from opacus.validators import ModuleValidator
import torch
from processor import DataProcessor
from logger_config import setup_logger


logger = setup_logger(__name__)


def generate_synthetic_data(df, sensitive, model_name: str = "ctgan", epochs: int = 300, batch_size: int = 500, cuda: bool = True, use_dp: bool = False, noise_multiplier: float = 1.3, max_grad_norm: float = 1.0, epsilon_delta: float = 1e-5):
    # разделение на train/test
    train_df, test_df = train_test_split(
        df,
        test_size=0.2,
        random_state=42,
        stratify=df[sensitive]
    )

    # добавляем 'id'
    train_df = train_df.reset_index(drop=True)
    train_df['id'] = train_df.index

    metadata = Metadata.detect_from_dataframe(data=train_df, table_name='adult')
    metadata.set_primary_key(table_name='adult', column_name='id')
    metadata.save_to_json(filepath='metadata.json', mode = 'overwrite')

    generators = {
        "ctgan": CTGANSynthesizer,
        "tvae": TVAESynthesizer,
        "copulagan": CopulaGANSynthesizer,
    }

    model = generators[model_name](
        metadata,
        epochs=epochs,
        batch_size=batch_size,
        verbose=True,
        cuda=cuda
        )

    model.fit(train_df)

    if use_dp:
        try:
            # Получаем дискриминатор, оптимизатор и загрузчик данных
            discriminator = model._model.discriminator
            optimizer = model._optimizer['discriminator']
            data_loader = model._data_loader

            # Проверка модели на совместимость с Opacus
            discriminator = ModuleValidator.fix(discriminator)
            errors = ModuleValidator.validate(discriminator, strict=False)
            if errors:
                logger.warning("[DP] Проблемы совместимости: %s", errors)

            # Инициализация механизма дифференциальной приватности
            privacy_engine = PrivacyEngine()
            discriminator, optimizer, data_loader = privacy_engine.make_private(
                module=discriminator,
                optimizer=optimizer,
                data_loader=data_loader,
                noise_multiplier=noise_multiplier,
                max_grad_norm=max_grad_norm
            )

            # Повторное краткое приватное дообучение
            discriminator.train()
            device = next(discriminator.parameters()).device
            for epoch in range(5):
                for x in data_loader:
                    x = x.to(device)
                    optimizer.zero_grad()
                    preds = discriminator(x)
                    loss = torch.mean(preds)  # упрощенная функция потерь
                    loss.backward()
                    optimizer.step()

            # Расчёт значения эпсилон
            epsilon, best_alpha = privacy_engine.get_epsilon(delta=epsilon_delta)
            logger.info(f"[DP] Эпсилон после {epochs}+5 эпох: {epsilon:.2f} (delta={epsilon_delta})")

        except Exception as e:
            logger.error("[DP] Ошибка при применении дифференциальной приватности: %s", e)

    syn_df = model.sample(num_rows=len(train_df)).drop(columns=['id'])
    # И сразу же биннинг + astype(str)
    dp = DataProcessor(syn_df)
    syn_df = dp.preprocess()
    return syn_df
